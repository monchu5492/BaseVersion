import { Component } from 'react';
import PropTypes from 'prop-types';
import '../../../stylesheets/AddUserForm.scss';
 
class AddUserForm extends Component {
  constructor(props) {
    super(props);
    this.submit = this.submit.bind(this);
  }
 
  submit(e) {
    const {
      _email, _firstname, _lastname, _password,
    } = this.refs;
    const { onNewUser } = this.props;
    e.preventDefault();
    onNewUser(_email.value, _firstname.value, _lastname.value, _password.value);
    _email.value = '';
    _firstname.value = '';
    _lastname.value = '';
    _password.value = '';
    _email.focus();
  }
 
  render() {
    return (
      <div id="Signup">
        <form action="http://localhost:4500/signup" method="post">
          <input name="email" type="email" className="form-control" placeholder="Username (email address)" required autoFocus />
          <input name="firstname" type="text" className="form-control" placeholder="First Name" required autoFocus />
          <input name="lastname" type="text" className="form-control" placeholder="Last Name" required autoFocus />
          <input name="password" type="password" className="form-control" placeholder="Password" required />
          <input type="submit" defaultValue="Sign up" className="btn btn-primary btn-block" />
        </form>
      </div>
    );
  }
}
 
AddUserForm.propTypes = {
  onNewUser: PropTypes.func,
};
 
AddUserForm.defaultProps = {
  onNewUser: f => f,
};
 
export default AddUserForm;
